  1. PCA for visualisation
# Remove target variable
df_features <- df %>% select(-Diabetes_binary)

# Standardize features
df_scaled <- scale(df_features)

# Run PCA
pca_result <- prcomp(df_scaled, center = TRUE, scale. = TRUE)

# Visualize PCA (first 2 components)
fviz_pca_ind(pca_result,
             geom.ind = "point",
             pointshape = 21,
             col.ind = as.factor(df$Diabetes_binary),
             palette = c("#00AFBB", "#FC4E07"),
             legend.title = "Diabetes") +
  labs(title = "PCA - BRFSS Diabetes Indicators")
